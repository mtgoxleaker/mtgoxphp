 645  	public static function parseVersion($v) {
 646          if ($v == 0) return '[unknown]';
 647          if ($v > 10000) {
 648            // Edits by ArtForz (Bitcoin Core Dev) fixing version checking

 649              // [22:06:18] <ArtForz> new is major * 10000 + minor * 100 + revision

 650              $rem = floor($v / 100);
 651              $proto = $v - ($rem*100);
 652              $v = $rem;
 653          } else {
 654              // [22:06:05] <ArtForz> old was major * 100 + minor

 655              $proto = 0;
 656          }
 657          foreach(array('revision','minor','major') as $type) {
 658              $rem = floor($v / 100);
 659              $$type = $v - ($rem * 100);
 660              $v = $rem;
 661          }
 662          // build string

 663          return $major . '.' . $minor . '.' . $revision . ($proto?('[.'.$proto.']'):'');
 664      }

1316     * Build transactions to transfer amount from list of inputs to output with remainder,

1317     * using a specified fee.

1318     *

1319     * Creates new Available_Output database entries for any intermediate transactions

1320     * which are generated.

1321     * 

1322     * @param input list of inputs to use

1323     * @param amount of output of final transaction

1324     * @param final_output address to output to

1325     * @param remainder address for remainder

1326     * @param fee to include in final transaction, default is 0

1327     * @return list of transactions to produce desired output

1328     */
1329  	public static function makeNormalTx($input, $amount, $final_output, $remainder, $fee = 0) {
1330          $res = array();
1331          
1332          // while there are more than 5 inputs, try to merge them

1333          while(count($input) > 5) {
1334              $xinput = array();
1335              $output = self::getNullAddr(true); // obtain new empty address

1336  
1337              // merge as many inputs as we can, while character length of the transaction < 1000

1338              while(true) {
1339                // move next input onto tinput array for checking length

1340                  $extra = array_shift($input);
1341                  if (is_null($extra)) break;
1342                  $tinput = $xinput;
1343                  $tinput[] = $extra;
1344                  $total = 0;
1345                  foreach($tinput as $t) $total+=$t['amount'];
1346                  // construct transaction to check its length

1347                  $ttx = \Util\Bitcoin::makeNormalTx($tinput, $total, $output['info'], $output['info']);
1348                  // stop when length of transaction for all of tinput inputs is greater than 1000

1349                  if (strlen($ttx) >= 1000) break;
1350                  $xinput[] = $extra;
1351              }
1352              if (!is_null($extra)) // put last input tested back
1353                  array_unshift($input, $extra);
1354                  
1355              $total = 0;
1356              foreach($xinput as $t) $total += $t['amount'];
1357              // construct transaction for the inputs on the xinput list

1358              $ttx = \Util\Bitcoin::makeNormalTx($xinput, $total, $output['info'], $output['info']);
1359              $res[] = $ttx;
1360              $thash = bin2hex(strrev(\Util\Bitcoin::hash_sha256(\Util\Bitcoin::hash_sha256($ttx))));
1361              // put this output on the list of now available inputs

1362              $input[] = array(
1363                  'amount' => $total,
1364                  'tx' => $thash,
1365                  'N' => 0,
1366                  'privkey' => $output['priv'],
1367                  'hash' => $output['info']['hash'],
1368              );
1369              // insert new Available_Output entry for this transaction, marked unavailable

1370              // possible leak here because these transactions have not yet been transmitted on the blockchain

1372          }

1375          $res[] = \Util\Bitcoin::makeNormalTx($input, $amount, $final_output, $remainder, $fee);
